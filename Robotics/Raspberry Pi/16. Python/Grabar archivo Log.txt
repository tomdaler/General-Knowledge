def  WriteLog(texto):
    path = "/var/log/"
    log = open(path + datetime.datetime.now().strftime("%Y-%m-%d") + "_dht.log","a")
    line = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S") + " " + texto + "\n"
    log.write(line)
    log.close()
